DELETE A NUMBER FROM LINKED LIST

CODE

void deleteNode(Node** head_ref, int key) {
    Node* temp = *head_ref;
    Node* prev = NULL;
    
    if (temp != NULL && temp->data == key) {
        *head_ref = temp->next;
        delete temp;
        return;
    } else {
        while (temp != NULL && temp->data != key) {
            prev = temp;
            temp = temp->next;
        }
        if (temp == NULL)
            return;
        prev->next = temp->next;
        delete temp;
    }
}

ALGORITHM

1. Create two pointers temp & prev temp points to head and prev points NULL.
2. Check if header value is equal to number or not.
3. If yes then shift header pointer to next node and return.
4. If not then check if next node value is equal to number.
5. If not then shift prev and temp.
6. If not present then return.                    
7. If value find then assign address of temp next node to prev next node and then delete temp.
